$import[] ="
    lib.controllers.core, 
    lib.models.mysql_connect,

    lib.config.config,
    lib.models.lib, 
    lib.models.thing, 
    lib.models.gpvar,

    lib.controllers.auth,
    lib.controllers.post,
    lib.controllers.tag,
    lib.controllers.setting,
    lib.controllers.page,
    lib.controllers.validator,
    lib.controllers.ticket,
    lib.controllers.view,
    lib.views.gallery,
    lib.views.ajaxfield";
    
    if (isset ($import) && sizeof ($import) > 0) {
        foreach ($import as $w) {
            foreach (
                explode (',', 
                    implode (',', 
                        array_map (
                            create_function (
                                '$z',
                                'return implode(",",glob("$z.php"));'
                            ), array_map ('trim',
                                explode (',',
                                    str_replace ('.', '/', $w)
                                )
                            )
                        )
                    )
                ) as $x) {
                    if ($x) require_once $x;
            }
        }
    }
